34      @Rule
35      public JenkinsRule jenkinsRule = new JenkinsRule();
36  
37      private static final String JENKINS_MASTER = "master";
38      private static final String JENKINS_TEST_JOB = "jenkins-client-test";
39  
40      private JenkinsServer server;
41  
42      @Before
43      public void setUp() throws Exception {
44          jenkinsRule.getInstance().getCrumbIssuer().getCrumb();
45          jenkinsRule.getInstance().getCrumbIssuer().getCrumbRequestField();
46          server = new JenkinsServer(jenkinsRule.getURL().toURI());
47      }
48  
49      @Test
50      public void shouldReturnListOfJobs() throws Exception {
51          jenkinsRule.getInstance().createProject(FreeStyleProject.class, JENKINS_TEST_JOB);
52          assertTrue(server.getJobs().containsKey(JENKINS_TEST_JOB));
53      }
54  
55      @Test
56      public void shouldReturnBuildsForJob() throws Exception {
57          FreeStyleProject trunk = jenkinsRule.getInstance().createProject(FreeStyleProject.class, JENKINS_TEST_JOB);
58          for (int i = 0; i < 5; i++)
59              trunk.scheduleBuild(0, new Cause.UserIdCause(),
60                      new ParametersAction(new StringParameterValue("BUILD NUMBER", "" + i)));
61  
62          while (trunk.isInQueue() || trunk.isBuilding()) {
63          }
64  
65          JobWithDetails job = server.getJobs().get(JENKINS_TEST_JOB).details();
66          assertEquals(5, job.getBuilds().get(0).getNumber());
67      }
68  
69      @Test
70      public void shouldReturnBuildStatusForBuild() throws Exception {
71          FreeStyleProject pr = jenkinsRule.getInstance().createProject(FreeStyleProject.class, JENKINS_TEST_JOB);
72  
73          pr.scheduleBuild2(0, new Cause.UserIdCause(),
74                  new ParametersAction(new StringParameterValue("REVISION", "foobar"))).get();
75  
76          JobWithDetails job = server.getJobs().get(JENKINS_TEST_JOB).details();
77          BuildWithDetails build = job.getBuilds().get(0).details();
78          assertEquals(BuildResult.SUCCESS, build.getResult());
79          assertEquals("foobar", build.getParameters().get("REVISION"));
80      }
81  
82      @Test
83      public void shouldSupportBooleanParameters() throws Exception {
84          FreeStyleProject pr = jenkinsRule.getInstance().createProject(FreeStyleProject.class, JENKINS_TEST_JOB);
85  
86          pr.scheduleBuild2(0, new Cause.UserIdCause(),
87                  new ParametersAction(new BooleanParameterValue("someValue", true))).get();
88  
89          JobWithDetails job = server.getJobs().get(JENKINS_TEST_JOB).details();
90          BuildWithDetails build = job.getBuilds().get(0).details();
91          assertEquals("true", build.getParameters().get("someValue"));
92      }

137     @Test
138     public void testCreateJob() throws Exception {
139         final String jobName = "test-job-" + UUID.randomUUID().toString();
140 
141         jenkinsRule.getInstance().createProject(FreeStyleProject.class, JENKINS_TEST_JOB);
142         String sourceXml = server.getJobXml(JENKINS_TEST_JOB);
143 
144         server.createJob(jobName, sourceXml);
145 
146         Map<String, Job> jobs = server.getJobs();
147         assertTrue(jobs.containsKey(jobName));
148         JobWithDetails thisJob = jobs.get(jobName).details();
149         assertNotNull(thisJob);
150         assertTrue(thisJob.getBuilds().size() == 0);
151     }
152 
153     @Test
154     public void shouldBuildAJob() throws Exception {
155         final String jobName = "test-job-" + UUID.randomUUID().toString();
156 
157         FreeStyleProject project = jenkinsRule.getInstance().createProject(FreeStyleProject.class, jobName);
158         assertTrue(jenkinsRule.getInstance().getJobNames().contains(jobName));
159 
160         JobWithDetails job = server.getJob(jobName);
161         job.build();
162 
163         while (project.isInQueue() || project.isBuilding()) {
164         }
165 
166         job = server.getJob(jobName);
167         assertTrue(job.getBuilds().size() == 1);
168     }
169 
170     @Test
171     public void testUpdateJob() throws Exception {
172         final String description = "test-" + UUID.randomUUID().toString();
173 
174         FreeStyleProject freeStyleProject = jenkinsRule.getInstance().createProject(FreeStyleProject.class,
175                 JENKINS_TEST_JOB);
176         freeStyleProject.setDescription(description);
177 
178         String sourceXml = server.getJobXml(JENKINS_TEST_JOB);
179         String newXml = sourceXml.replaceAll("<description>.*</description>",
180                 "<description>" + description + "</description>");
181         server.updateJob(JENKINS_TEST_JOB, newXml);
182 
183         String confirmXml = server.getJobXml(JENKINS_TEST_JOB);
184         assertTrue(confirmXml.contains(description));
185     }
