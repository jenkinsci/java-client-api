7   package com.offbytwo.jenkins.model;
8   
9   public class Artifact extends BaseModel {
10  
11      String displayPath;
12      String fileName;
13      String relativePath;
14  
15      public String getDisplayPath() {
16          return displayPath;
17      }
18  
19      public void setDisplayPath(String displayPath) {
20          this.displayPath = displayPath;
21      }
22  
23      public String getFileName() {
24          return fileName;
25      }
26  
27      public void setFileName(String fileName) {
28          this.fileName = fileName;
29      }
30  
31      public String getRelativePath() {
32          return relativePath;
33      }
34  
35      public void setRelativePath(String relativePath) {
36          this.relativePath = relativePath;
37      }
38  
39      @Override
40      public boolean equals(Object o) {
41          if (this == o) return true;
42          if (o == null || getClass() != o.getClass()) return false;
43  
44          Artifact artifact = (Artifact) o;
45  
46          if (displayPath != null ? !displayPath.equals(artifact.displayPath) : artifact.displayPath != null)
47              return false;
48          if (fileName != null ? !fileName.equals(artifact.fileName) : artifact.fileName != null)
49              return false;
50          if (relativePath != null ? !relativePath.equals(artifact.relativePath) : artifact.relativePath != null)
51              return false;
52  
53          return true;
54      }
55  
56      @Override
57      public int hashCode() {
58          int result = displayPath != null ? displayPath.hashCode() : 0;
59          result = 31 * result + (fileName != null ? fileName.hashCode() : 0);
60          result = 31 * result + (relativePath != null ? relativePath.hashCode() : 0);
61          return result;
62      }
63  }
