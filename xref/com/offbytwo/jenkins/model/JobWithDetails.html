18  public class JobWithDetails extends Job {
19  
20      String displayName;
21      boolean buildable;
22      List<Build> builds;
23      Build firstBuild;
24      Build lastBuild;
25      Build lastCompletedBuild;
26      Build lastFailedBuild;
27      Build lastStableBuild;
28      Build lastSuccessfulBuild;
29      Build lastUnstableBuild;
30      Build lastUnsuccessfulBuild;
31      int nextBuildNumber;
32      List<Job> downstreamProjects;
33      List<Job> upstreamProjects;

35      public String getDisplayName() {
36          return displayName;
37      }
38  
39      public boolean isBuildable() {
40          return buildable;
41      }
42  
43      public List<Build> getBuilds() {
44          return transform(builds, new Function<Build, Build>() {
45              @Override
46              public Build apply(Build from) {
47                  return buildWithClient(from);
48              }
49          });
50      }
51  
52      private Build buildWithClient(Build from) {
53          Build ret = from;
54          if (from != null) {
55              ret = new Build(from);
56              ret.setClient(client);
57          }
58          return ret;
59      }
60

62          return firstBuild;
63      }
64  
65      public Build getLastBuild() {
66          return buildWithClient(lastBuild);
67      }

70          return buildWithClient(lastCompletedBuild);

72  
73      public Build getLastFailedBuild() {
74          return buildWithClient(lastFailedBuild);
75      }

77      public Build getLastStableBuild() {

87      }
88

105     @Override
106     public boolean equals(Object o) {
107         if (this == o) return true;
108         if (o == null || getClass() != o.getClass()) return false;
109         if (!super.equals(o)) return false;
110 
111         JobWithDetails that = (JobWithDetails) o;
112 
113         if (buildable != that.buildable) return false;
114         if (nextBuildNumber != that.nextBuildNumber) return false;
115         if (builds != null ? !builds.equals(that.builds) : that.builds != null)
116             return false;
117         if (displayName != null ? !displayName.equals(that.displayName) : that.displayName != null)
118             return false;
119         if (downstreamProjects != null ? !downstreamProjects.equals(that.downstreamProjects) : that.downstreamProjects != null)
120             return false;
121         if (lastBuild != null ? !lastBuild.equals(that.lastBuild) : that.lastBuild != null)
122             return false;
123         if (lastCompletedBuild != null ? !lastCompletedBuild.equals(that.lastCompletedBuild) : that.lastCompletedBuild != null)
124             return false;
125         if (lastFailedBuild != null ? !lastFailedBuild.equals(that.lastFailedBuild) : that.lastFailedBuild != null)
126             return false;
127         if (lastStableBuild != null ? !lastStableBuild.equals(that.lastStableBuild) : that.lastStableBuild != null)
128             return false;
129         if (lastSuccessfulBuild != null ? !lastSuccessfulBuild.equals(that.lastSuccessfulBuild) : that.lastSuccessfulBuild != null)
130             return false;
131         if (lastUnstableBuild != null ? !lastUnstableBuild.equals(that.lastUnstableBuild) : that.lastUnstableBuild != null)
132             return false;

141 	public Build getBuildByNumber(final int buildNumber) {
142         
143     Predicate<Build> isMatchingBuildNumber = new Predicate<Build>() {
144             
145             @Override
146             public boolean apply(Build input) {
147                 return input.getNumber() == buildNumber;
148             }
149         };
150         
151         Optional<Build> optionalBuild = Iterables.tryFind(builds, isMatchingBuildNumber);
152         return optionalBuild.orNull() == null ? null : buildWithClient(optionalBuild.orNull());
153     }

155     @Override
156     public int hashCode() {
157         int result = super.hashCode();
158         result = 31 * result + (displayName != null ? displayName.hashCode() : 0);
159         result = 31 * result + (buildable ? 1 : 0);
160         result = 31 * result + (builds != null ? builds.hashCode() : 0);
161         result = 31 * result + (lastBuild != null ? lastBuild.hashCode() : 0);
162         result = 31 * result + (lastCompletedBuild != null ? lastCompletedBuild.hashCode() : 0);
163         result = 31 * result + (lastFailedBuild != null ? lastFailedBuild.hashCode() : 0);
164         result = 31 * result + (lastStableBuild != null ? lastStableBuild.hashCode() : 0);
165         result = 31 * result + (lastSuccessfulBuild != null ? lastSuccessfulBuild.hashCode() : 0);
166         result = 31 * result + (lastUnstableBuild != null ? lastUnstableBuild.hashCode() : 0);
167         result = 31 * result + (lastUnsuccessfulBuild != null ? lastUnsuccessfulBuild.hashCode() : 0);
168         result = 31 * result + nextBuildNumber;
169         result = 31 * result + (downstreamProjects != null ? downstreamProjects.hashCode() : 0);
170         result = 31 * result + (upstreamProjects != null ? upstreamProjects.hashCode() : 0);
171         return result;
172     }
