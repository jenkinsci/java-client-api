25      public JobConfiguration(String configXml) throws DocumentException {
26          this.configXml = configXml;
27          reader = new SAXReader();
28          doc = reader.read(new StringReader(configXml));

31      public JobConfiguration addStringParam(String name, String desc, String defaultValue) throws JAXBException, DocumentException {
32          List<Node> nodes = doc.selectNodes("//hudson.model.ParametersDefinitionProperty");
33          StringWriter sw = new StringWriter();
34          StringParameterDefinition spd = new StringParameterDefinition(name, desc, defaultValue);
35          if (null == nodes || 0 == nodes.size()) {
36              ParameterDefinitions pd = new ParameterDefinitions();
37              pd.addParam(spd);
38              ParametersDefinitionProperty pdp = new ParametersDefinitionProperty(pd);
39              JAXBContext jaxbContext = JAXBContext.newInstance(ParametersDefinitionProperty.class);
40              Marshaller marshaller = jaxbContext.createMarshaller();
41              marshaller.setProperty(Marshaller.JAXB_FRAGMENT, true);
42              marshaller.setProperty(Marshaller.JAXB_FORMATTED_OUTPUT, true);
43              marshaller.marshal(pdp, sw);
44              Document docInterlude = reader.read(new StringReader(sw.toString()));
45              List<Node> propertiesNode = doc.selectNodes("//project/properties");
46              for (Node node : propertiesNode) {
47                  if (node instanceof Element) {
48                      Element e = (Element) node;
49                      e.add(docInterlude.getRootElement());
50                  }
51              }
52          } else {
53              JAXBContext jaxbContext = JAXBContext.newInstance(StringParameterDefinition.class);
54              Marshaller marshaller = jaxbContext.createMarshaller();
55              marshaller.setProperty(Marshaller.JAXB_FRAGMENT, true);
56              marshaller.setProperty(Marshaller.JAXB_FORMATTED_OUTPUT, true);
57              marshaller.marshal(spd, sw);
58              Document docInterlude = reader.read(new StringReader(sw.toString()));
59              List<Node> propertiesNode = doc.selectNodes("//parameterDefinitions");
60              for (Node node : propertiesNode) {
61                  if (node instanceof Element) {
62                      Element e = (Element) node;
63                      e.add(docInterlude.getRootElement());
64                  }
65              }
66          }
67          return this;
68  
69      }
