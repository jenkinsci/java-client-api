1   package com.offbytwo.jenkins.client.util;

3   import java.io.FilterInputStream;
4   import java.io.IOException;
5   import java.io.InputStream;
6   
7   import org.apache.http.client.methods.HttpRequestBase;
8   
9   /**
10   * Releases the http request method if it is closed
11   */
12  public class RequestReleasingInputStream extends FilterInputStream {
13      private final HttpRequestBase httpRequestBase;
14  
15      /**
16       * Creates a <code>FilterInputStream</code>
17       * by assigning the  argument <code>in</code>
18       * to the field <code>this.in</code> so as
19       * to remember it for later use.
20       *
21       * @param in              the underlying input stream, or <code>null</code> if
22       *                        this instance is to be created without an underlying stream.
23       * @param httpRequestBase The request object that should be released if the stream closed
24       */
25      public RequestReleasingInputStream(InputStream in, HttpRequestBase httpRequestBase) {
26          super(in);
27          this.httpRequestBase = httpRequestBase;
28      }
29  
30      /**
31       * Closes this input stream and releases any system resources
32       * associated with the stream.
33       * This
34       * method simply performs <code>in.close()</code>.
35       *
36       * @throws IOException if an I/O error occurs.
37       * @see FilterInputStream#in
38       */
39      @Override public void close() throws IOException {
40          super.close();
41          httpRequestBase.releaseConnection();
42  
43      }
44  }
