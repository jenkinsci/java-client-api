10      public static String encode(String pathPart) {
11          // jenkins doesn't like the + for space, use %20 instead
12          String escape = UrlEscapers.urlPathSegmentEscaper().escape(pathPart);
13          return escape;
14      }
15  
16      public static String encodeParam(String pathPart) {
17          // jenkins doesn't like the + for space, use %20 instead
18          return UrlEscapers.urlFormParameterEscaper().escape(pathPart);
19      }
