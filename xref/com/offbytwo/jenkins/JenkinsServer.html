127     /**
128      * Get a list of all the defined jobs on the server (at the specified view
129      * level and in the specified folder)
130      *
131      * @return list of defined jobs (view level, for details @see Job#details
132      * @throws IOException
133      */
134     public Map<String, Job> getJobs(FolderJob folder, String view) throws IOException {
135         String path = "/";
136         if (folder != null) {
137             path = folder.getUrl();
138         }
139         Class<? extends MainView> viewClass = MainView.class;
140         if (view != null) {
141             path = path + "view/" + EncodingUtils.encode(view) + "/";
142             viewClass = View.class;
143         }
144         List<Job> jobs = client.get(path, viewClass).getJobs();
145         return Maps.uniqueIndex(jobs, new Function<Job, String>() {
146             @Override
147             public String apply(Job job) {
148                 job.setClient(client);
149                 return job.getName().toLowerCase();
150             }
151         });
152     }

270     public Optional<FolderJob> getFolderJob(Job job) throws IOException {
271         try {
272             FolderJob folder = client.get(job.getUrl(), FolderJob.class);
273             if (!folder.isFolder()) {
274                 return Optional.absent();
275             }
276             folder.setClient(client);
277 
278             return Optional.of(folder);
279         } catch (HttpResponseException e) {
280             if (e.getStatusCode() == HttpStatus.SC_NOT_FOUND) {
281                 return null;
282             }
283             throw e;
284         }

374         client.post_form(path + "createItem?", params, crumbFlag);
375     }

441     public void addStringParam(String jobName, String name, String description, String defaultValue)
442             throws IOException, JAXBException, DocumentException {
443         String jobXml = this.getJobXml(jobName);
444         JobConfiguration jobConf = new JobConfiguration(jobXml);
445         jobXml = jobConf.addStringParam(name, description, defaultValue).asXml();
446         this.updateJob(jobName, jobXml);
447     }

499     /**
500      * Runs the provided groovy script on the server and returns the result.
501      *
502      * This is similar to running groovy scripts using the script console.
503      *
504      * In the instance where your script causes an exception, the server still
505      * returns a 200 status, so detecting errors is very challenging. It is
506      * recommended to use heuristics to check your return string for stack
507      * traces by detecting strings like "groovy.lang.(something)Exception".
508      *
509      * @param script
510      * @return results
511      * @throws IOException
512      */
513     public String runScript(String script) throws IOException {
514         return client.post_text("/scriptText", "script=" + script, ContentType.APPLICATION_FORM_URLENCODED, false);
515     }
