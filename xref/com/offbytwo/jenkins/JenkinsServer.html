78      /**
79       * Get the current status of the Jenkins end-point by pinging it.
80       *
81       * @return true if Jenkins is up and running, false otherwise

84          try {

92      /**
93       * Get a list of all the defined jobs on the server (at the summary level)
94       *
95       * @return list of defined jobs (summary level, for details @see Job#details
96       * @throws IOException
97       */
98      public Map<String, Job> getJobs() throws IOException {
99          List<Job> jobs = client.get("/", MainView.class).getJobs();
100         return Maps.uniqueIndex(jobs, new Function<Job, String>() {
101             @Override
102             public String apply(Job job) {
103                 job.setClient(client);
104                 return job.getName().toLowerCase();
105             }
106         });
107     }

270     public void addStringParam(String jobName, String name, String description, String defaultValue)
271             throws IOException, JAXBException, DocumentException {
272         String jobXml = this.getJobXml(jobName);
273         JobConfiguration jobConf = new JobConfiguration(jobXml);
274         jobXml = jobConf.addStringParam(name, description, defaultValue).asXml();
275         this.updateJob(jobName, jobXml);
276     }

328     /**
329      * Runs the provided groovy script on the server and returns the result.
330      *
331      * This is similar to running groovy scripts using the script console.
332      *
333      * In the instance where your script causes an exception, the server still
334      * returns a 200 status, so detecting errors is very challenging. It is
335      * recommended to use heuristics to check your return string for stack
336      * traces by detecting strings like "groovy.lang.(something)Exception".
337      *
338      * @param script
339      * @return results
340      * @throws IOException
341      */
342     public String runScript(String script) throws IOException {
343         return client.post_text("/scriptText", "script=" + script, ContentType.APPLICATION_FORM_URLENCODED, false);
344     }

346     private String encode(String pathPart) {
347         // jenkins doesn't like the + for space, use %20 instead
348         String escape = UrlEscapers.urlPathSegmentEscaper().escape(pathPart);
349         return escape;
350     }
351 
352     private String encodeParam(String pathPart) {
353         // jenkins doesn't like the + for space, use %20 instead
354         return UrlEscapers.urlFormParameterEscaper().escape(pathPart);
355     }
